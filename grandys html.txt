function checkGmail(){

let email=document.getElementById("gmail").value;

if(!email.endsWith("@gmail.com")){
alert("Error: Please enter a Gmail address ending with @gmail.com");
return;
}

document.getElementById("gmailLogin").style.display="none";
document.getElementById("passwordLogin").style.display="flex";

}

function passwordLogin(){

let pass=document.getElementById("grandypass").value;

let saved=localStorage.getItem("grandysPassword");

if(saved==null){

localStorage.setItem("grandysPassword",pass);
alert("Password created successfully!");

showSite();

}else{

if(pass===saved){
showSite();
}else{
alert("Wrong Grandys Password!");
}

}

}

function showSite(){

document.getElementById("passwordLogin").style.display="none";
document.getElementById("site").style.display="block";

}
